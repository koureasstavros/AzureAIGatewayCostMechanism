let requestCorr = customMetrics
    | where timestamp > startofmonth(now())
    | where name == "Request Correlation"
    | extend
        userId           = tostring(customDimensions["User ID"]),
        userEmail        = tostring(customDimensions["User Email"]),
        productId        = tostring(customDimensions["Product ID"]),
        productName      = tostring(customDimensions["Product Name"]),
        subscriptionId   = tostring(customDimensions["Subscription ID"]),
        subscriptionName = tostring(customDimensions["Subscription Name"]),
        apiId            = tostring(customDimensions["API ID"]),
        operationId      = tostring(customDimensions["Operation ID"])
    | project userId, userEmail, productId, productName, subscriptionId, subscriptionName, apiId, operationId
    | distinct userId, userEmail, productId, productName, subscriptionId, subscriptionName, apiId, operationId;
let costCorr = customMetrics
    | where timestamp > startofmonth(now())
    | where name == "Cost Correlation"
    | extend
        subscriptionId  = tostring(customDimensions["Subscription ID"]),
        apiId           = tostring(customDimensions["API ID"]),
        operationId     = tostring(customDimensions["Operation ID"]),
        modelId         = tostring(customDimensions["Model ID"]),
        statusCode      = tostring(customDimensions["Status Code"])
    | project subscriptionId, apiId, operationId, modelId, statusCode, value;
requestCorr
| join kind=inner costCorr on subscriptionId, apiId, operationId
| summarize
    Cost         = sum(value)
    by userEmail, productName, subscriptionName, apiId, modelId, statusCode
| extend QuotaCost = 100
| extend RatioCost = (Cost / QuotaCost) * 100
| project userEmail, productName, subscriptionName, apiId, modelId, statusCode, Cost, QuotaCost, RatioCost;
